authenticate

)
Function

Defined in lib/bella.js:169

Method responsible for authenticating access the API. Only users with Access Token, Domain and IP authenticated can access the API.

Parameters:

  • permission String

    The required permission

Returns:

Function:

Returns a middleware

Example:

app.configure(function() {
  app.use(bella(mongoose, {connection: mongooseConnection, status: false, model: userModel}));
  app.use(bella.authenticate());
});

//OR

app.get('/users', bella.authenticate('user'), ctrl);
